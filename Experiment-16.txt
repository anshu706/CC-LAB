Binary tree is symmetric
Problem Statement:

Write a program to verify and validate if a tree is mirrored (symmetric) or not.

A tree is symmetric if its left subtree is a mirror image of its right subtree. This means:

The root's left child's value must equal the root's right child's value.
The left child's left subtree must be a mirror of the right child's right subtree.
The left child's right subtree must be a mirror of the right child's left subtree.
This suggests a recursive helper function that takes two nodes, t1 and t2, and checks if they are mirror images of each other.

Input Explanation:

The first input is an integer n representing the number of nodes in the tree.
The second input is a sequence of n integers where each integer represents the value of a node in the tree in level order (from top to bottom, left to right).
The value -1 represents a null (empty) node in the level-order input.
Output Explanation:

The output is a single boolean value (true or false):

true if the binary tree is symmetric (mirror image of itself).
false if the binary tree is not symmetric.
Constraints:

The number of nodes in the tree is in the range [1, 1000].
-100 <= Node.val <= 100
Time Limit:
1 sec

Example 1:
Input
7
1 2 2 3 4 4 3

Output
True

Example 2:
Input
5
1 2 2 -1 3 -1 3

Output
False

Sample I/O Explanation:

Explanation:

Test case 1:

This tree is symmetric because the left subtree (2 -> 3, 4) is a mirror image of the right subtree (2 -> 4, 3).



Test Case 2:

This tree is not symmetric because the left subtree (2 -> 3) does not mirror the right subtree (2 -> 3). Specifically, the left subtree has a right child, while the right subtree has a right child too, but they don't mirror each other.

------------- C++ -----------------------

#include <iostream>
#include <vector>
using namespace std;

class SymmetricTree {
public:
    struct TreeNode {
        int val;
        TreeNode* left;
        TreeNode* right;
        TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
    };

    // Helper function to check if two trees are mirror images
    bool isMirror(TreeNode* t1, TreeNode* t2) {
        if (t1 == nullptr && t2 == nullptr) return true;
        if (t1 == nullptr || t2 == nullptr) return false;
        return (t1->val == t2->val) &&
               isMirror(t1->left, t2->right) &&
               isMirror(t1->right, t2->left);
    }

    // Main function to check symmetry
    bool isSymmetric(TreeNode* root) {
        if (root == nullptr) return true;
        return isMirror(root->left, root->right);
    }

    // Build tree from level-order input
    static TreeNode* buildTree(const vector<int>& values) {
        if (values.empty() || values[0] == -1) {
            return nullptr;
        }

        TreeNode* root = new TreeNode(values[0]);
        vector<TreeNode*> nodes(values.size());
        nodes[0] = root;

        int index = 1;
        for (int i = 0; i < values.size(); i++) {
            if (nodes[i] == nullptr) continue;
            if (index < values.size() && values[index] != -1) {
                nodes[i]->left = new TreeNode(values[index]);
                nodes[index] = nodes[i]->left;
            }
            index++;
            if (index < values.size() && values[index] != -1) {
                nodes[i]->right = new TreeNode(values[index]);
                nodes[index] = nodes[i]->right;
            }
            index++;
        }
        return root;
    }
};

// ------------------ MAIN ------------------
int main() {
    int n;
    cin >> n;

    vector<int> values(n);
    for (int i = 0; i < n; i++) {
        cin >> values[i];
    }

    SymmetricTree checker;
    SymmetricTree::TreeNode* root = SymmetricTree::buildTree(values);
    cout << (checker.isSymmetric(root) ? "True" : "False") << endl;

    return 0;
}

-------------------------- java --------------------------------------

class TreeNode {
    int val;
    TreeNode left, right;

    TreeNode(int x) {
        val = x;
    }
}

public class SymmetricTree {

    static boolean isMirror(TreeNode a, TreeNode b) {
        if (a == null && b == null) return true;
        if (a == null || b == null) return false;

        return (a.val == b.val) &&
               isMirror(a.left, b.right) &&
               isMirror(a.right, b.left);
    }

    static boolean isSymmetric(TreeNode root) {
        if (root == null) return true;
        return isMirror(root.left, root.right);
    }

    public static void main(String[] args) {
        TreeNode root = new TreeNode(1);
        root.left = new TreeNode(2);
        root.right = new TreeNode(2);
        root.left.left = new TreeNode(3);
        root.right.right = new TreeNode(3);

        System.out.println(isSymmetric(root));
    }
}
