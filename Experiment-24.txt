Binary Tree Left View
Problem Statement:

You are working on a tree visualization module for a data-structure learning platform.
Given a binary tree, your task is to determine the Left View of the tree.
The Left View of a binary tree is the set of nodes visible when the tree is viewed from the left side.
At each level of the tree, only the first node encountered (from left to right) is included in the left view.
Write a program to compute and return the left view of the given binary tree.


Input Explanation:

An integer n representing the number of nodes in the binary tree
A level-order representation of the binary tree where -1 represents a null node


Output Explanation:

A list of integers representing the Left View of the binary tree, printed in order from top to bottom

Constraints:

0 ≤ n ≤ 10⁴
−10⁴ ≤ Node.val ≤ 10⁴


Time Limit:
1 sec

Example 1:
Input
5
1 2 3 4 5

Output
[1, 2, 4]

Sample I/O Explanation:

For the tree:
   1
   / \
  2  3
 /      \
5       4


Level-wise first visible nodes:
Level 0 → 1
Level 1 → 2
Level 2 → 5
Left View = [1, 2, 5]

==============================================================================================================================

import java.util.*;
class TreeNode {
    int val;
    TreeNode left, right;
    TreeNode(int x) {
        val = x;
        left = right = null;
    }
}
public class TreeLeftView {
    public static List<Integer> leftView(TreeNode root) {
        List<Integer> result = new ArrayList<>();
        if (root == null) return result;
        
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        
        while (!queue.isEmpty()) {
            int levelSize = queue.size();
            for (int i = 0; i < levelSize; i++) {
                TreeNode curr = queue.poll();
                if (i == 0) result.add(curr.val);
                if (curr.left != null) queue.add(curr.left);
                if (curr.right != null) queue.add(curr.right);
            }
        }
        return result;
    }
    // Do not edit this part
    private static TreeNode buildTree(int[] nodes) {
        if (nodes.length == 0 || nodes[0] == -1)
            return null;
        TreeNode root = new TreeNode(nodes[0]);
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        int i = 1;
        while (!queue.isEmpty() && i < nodes.length) {
            TreeNode curr = queue.poll();
            if (i < nodes.length && nodes[i] != -1) {
                curr.left = new TreeNode(nodes[i]);
                queue.add(curr.left);
            }
            i++;
            if (i < nodes.length && nodes[i] != -1) {
                curr.right = new TreeNode(nodes[i]);
                queue.add(curr.right);
            }
            i++;
        }
        return root;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] nodes = new int[n];
        for (int i = 0; i < n; i++) {
            nodes[i] = sc.nextInt();
        }
        TreeNode root = buildTree(nodes);
        List<Integer> result = leftView(root);
        System.out.print("[");
        for (int i = 0; i < result.size(); i++) {
            System.out.print(result.get(i));
            if (i != result.size() - 1)
                System.out.print(", ");
        }
        System.out.print("]");
        sc.close();
    }
}
