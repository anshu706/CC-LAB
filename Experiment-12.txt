Validate a Binary Search Tree
Problem Statement:

Objective(s)
To understand the properties of a Binary Search Tree (BST) and implement a function isValidBST() that checks whether a given binary tree follows BST rules.

Outcome
Students will be able to:

Implement a correct and efficient BST validation algorithm
Understand the difference between local checks and global range constraints
Apply recursion while updating min/max allowable ranges
Avoid common mistakes made in BST validation
Problem Statement
Write a program that validates whether a given binary tree is a Binary Search Tree (BST) using the function:

Input Explanation:

n

v1 v2 v3 ... vn

Output Explanation:

true or false

Constraints:

Number of nodes: 0 ≤ n ≤ 10⁵

Node values: any integers

Tree input is given in level-order format

Time Limit:
1 sec

Example 1:
Input
7
5 3 7 2 4 6 8

Output
true

Example 2:
Input
5
5 1 7 -1 6

Output
false

Sample I/O Explanation:

==============================================================================================================================


import java.util.*;

class TreeNode {
    int val;
    TreeNode left, right;
    TreeNode(int val) {
        this.val = val;
    }
}

public class ValidateBST {

    public static boolean isValidBST(TreeNode root) {
        return validate(root, Long.MIN_VALUE, Long.MAX_VALUE);
    }

    private static boolean validate(TreeNode node, long min, long max) {
        if (node == null) return true;
        if (node.val <= min || node.val >= max) return false;
        return validate(node.left, min, node.val) && validate(node.right, node.val, max);
    }

    // ------------------ BUILD TREE FROM LEVEL ORDER ------------------
    public static TreeNode buildTree(Integer[] arr) {
        if (arr.length == 0 || arr[0] == null) return null;
        TreeNode root = new TreeNode(arr[0]);
        Queue<TreeNode> q = new LinkedList<>();
        q.add(root);
        int i = 1;
        while (!q.isEmpty() && i < arr.length) {
            TreeNode curr = q.poll();
            if (arr[i] != null) {
                curr.left = new TreeNode(arr[i]);
                q.add(curr.left);
            }
            i++;
            if (i < arr.length && arr[i] != null) {
                curr.right = new TreeNode(arr[i]);
                q.add(curr.right);
            }
            i++;
        }
        return root;
    }

    // ------------------ MAIN FOR USER INPUT ------------------
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        Integer[] arr = new Integer[n];
        for (int i = 0; i < n; i++) {
            int val = sc.nextInt();
            arr[i] = (val == -1) ? null : val;
        }
        TreeNode root = buildTree(arr);
        System.out.println(isValidBST(root));
    }
}
